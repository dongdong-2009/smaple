 SPFieldChoice choiceField;        public List<Response> GetSurveyResponse()        {            List<Response> responseList;            string questionName = "";            SPWeb oSPWeb = SPContext.Current.Site.RootWeb;            SPList surveyList = oSPWeb.Lists["MySurvey1"];            if (surveyList.BaseType != SPBaseType.Survey)            {                throw new Exception("List is not of survey type");            }            responseList = new List<Response>();            foreach (SPField field in surveyList.Fields)            {                if (field.Type == SPFieldType.Choice)                {                    choiceField = (SPFieldChoice)field;                    questionName = field.StaticName;                    surveyQuestion = field.Title;                    Response resField = new Response();                    resField.Id = field.Id.ToString();                    resField.QuestionName = field.Title;                    responseList.Add(resField);                                  }            }            foreach (Response res in responseList)            {                SPQuery responseQuery = new SPQuery();                responseQuery.Query = @"<Where><Eq><FieldRef Name=" + questionName + " /><Value Type=\"Choice\">" + res.ChoiceName + @"</Value></Eq></Where>";                res.TotalResponse = surveyList.GetItems(responseQuery).Count;            }            return responseList;        }
